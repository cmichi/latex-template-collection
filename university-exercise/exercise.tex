\mysection{Aufgabe 4-1: Gruppierungsfunktionen}

\mysubsection{1}{SQL-Abfrage}
\begin{lstlisting}
SELECT
	TeileNr, Farbe, MIN(Preis) AS niedrigster_preis,
	MAX(Preis) AS hoechster_preis,
	MAX(Preis) - MIN(Preis) AS Differenz
FROM
	Liefert
WHERE
	LiefNr <> 0
GROUP BY
	TeileNr, Farbe
\end{lstlisting}
